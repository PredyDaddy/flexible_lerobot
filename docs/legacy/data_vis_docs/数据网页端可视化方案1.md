# LeRobot 数据网页端可视化方案

> **状态说明 (lerobot 0.4.3 baseline)**  
> 本文档描述的是“新增 Web 可视化服务”的方案。当前仓库 0.4.3 baseline 并不包含 `vis/` 实现与 `vis.sh`，因此本文不提供可直接执行的步骤。  
> 在路线 A 下，建议优先使用内置的 `lerobot-dataset-viz` 完成数据集可视化与质检。  

## 1. 项目背景与目标

### 1.1 现状分析

当前数据集可视化依赖 `vis.sh` 脚本，通过 `lerobot-dataset-viz` 命令逐条查看数据：

```bash
# vis.sh 当前内容
export HF_HUB_OFFLINE=1
lerobot-dataset-viz \
    --repo-id cqy/agilex_vla_demo_ee_pinocchio \
    --root /home/agilex/.cache/huggingface/lerobot/cqy/agilex_vla_demo_ee_pinocchio \
    --episode-index 0
```

**存在的问题：**
- 每次只能查看单个 episode，需要手动修改 `--episode-index` 参数
- 依赖 Rerun SDK 桌面应用，无法远程便捷访问
- 无法快速浏览整个数据集的概览信息
- 缺乏交互式的数据筛选和导航功能

### 1.2 目标

构建一个**独立的 Web 可视化模块**，实现：

1. **浏览器访问**：通过 URL 直接在浏览器中查看数据集
2. **数据集概览**：展示数据集元信息、统计数据、episode 列表
3. **Episode 浏览**：支持快速切换和播放任意 episode
4. **多相机视图**：同时展示所有相机画面
5. **数据叠加**：在视频上叠加 action/state 曲线
6. **不修改源码**：作为独立模块 `vis/` 存在

## 5. 核心实现代码

### 5.1 数据集管理器 (`core/dataset_manager.py`)

```python
"""数据集管理器 - 负责数据集的发现、加载和缓存"""
import json
from pathlib import Path
from typing import Optional
from functools import lru_cache

import pandas as pd
import pyarrow.parquet as pq

# 复用 LeRobot 的工具函数
from lerobot.datasets.utils import (
    load_info, load_stats, load_tasks, load_episodes,
    INFO_PATH, DEFAULT_VIDEO_PATH
)


class DatasetManager:
    """数据集管理器，提供数据集发现和元数据访问"""

    def __init__(self, root: Path):
        self.root = Path(root)
        self._cache = {}

    def discover_datasets(self) -> list[dict]:
        """发现根目录下所有有效的 LeRobot 数据集"""
        datasets = []
        # 遍历 user/repo_name 结构
        for user_dir in self.root.iterdir():
            if not user_dir.is_dir():
                continue
            for repo_dir in user_dir.iterdir():
                info_path = repo_dir / INFO_PATH
                if info_path.exists():
                    info = load_info(repo_dir)
                    datasets.append({
                        "repo_id": f"{user_dir.name}/{repo_dir.name}",
                        "path": str(repo_dir),
                        "total_episodes": info.get("total_episodes", 0),
                        "total_frames": info.get("total_frames", 0),
                        "fps": info.get("fps", 30),
                        "robot_type": info.get("robot_type", "unknown")
                    })
        return datasets

    @lru_cache(maxsize=10)
    def get_dataset_info(self, repo_id: str) -> dict:
        """获取数据集详细信息"""
        dataset_path = self.root / repo_id
        info = load_info(dataset_path)
        tasks = load_tasks(dataset_path)

        # 提取相机键
        camera_keys = [
            k for k, v in info.get("features", {}).items()
            if v.get("dtype") == "video" or "images" in k
        ]

        return {
            "repo_id": repo_id,
            **info,
            "camera_keys": camera_keys,
            "tasks": tasks.to_dict("records") if tasks is not None else []
        }

    def get_episodes(self, repo_id: str) -> list[dict]:
        """获取数据集的所有 episode 信息"""
        dataset_path = self.root / repo_id
        info = load_info(dataset_path)
        episodes_ds = load_episodes(dataset_path)
        tasks = load_tasks(dataset_path)
        fps = info.get("fps", 30)

        # 构建 task_index -> task 映射
        task_map = {}
        if tasks is not None:
            task_map = dict(zip(tasks["task_index"], tasks["task"]))

        episodes = []
        for i in range(len(episodes_ds)):
            ep = episodes_ds[i]
            from_idx = ep["dataset_from_index"]
            to_idx = ep["dataset_to_index"]
            num_frames = to_idx - from_idx
            episodes.append({
                "episode_index": i,
                "from_index": from_idx,
                "to_index": to_idx,
                "num_frames": num_frames,
                "task_index": ep.get("task_index", 0),
                "task": task_map.get(ep.get("task_index", 0), ""),
                "duration_s": round(num_frames / fps, 2)
            })
        return episodes
```

### 5.2 视频流处理器 (`core/video_streamer.py`)

```python
"""视频流处理器 - 负责视频文件定位和流式传输"""
from pathlib import Path
from typing import Optional
import io

from lerobot.datasets.utils import load_info, load_episodes, DEFAULT_VIDEO_PATH


class VideoStreamer:
    """视频流处理器"""

    def __init__(self, dataset_path: Path):
        self.dataset_path = Path(dataset_path)
        self.info = load_info(self.dataset_path)
        self.episodes = load_episodes(self.dataset_path)

    def get_video_path(self, camera_key: str, episode_index: int) -> Optional[Path]:
        """获取指定 episode 和相机的视频文件路径"""
        if episode_index >= len(self.episodes):
            return None

        ep = self.episodes[episode_index]
        # 从 episode 元数据中获取视频文件位置
        video_key = f"videos/{camera_key}"
        chunk_idx = ep.get(f"{video_key}/chunk_index", 0)
        file_idx = ep.get(f"{video_key}/file_index", 0)

        video_path = self.dataset_path / DEFAULT_VIDEO_PATH.format(
            video_key=camera_key,
            chunk_index=chunk_idx,
            file_index=file_idx
        )

        return video_path if video_path.exists() else None

    def get_video_info(self, camera_key: str, episode_index: int) -> dict:
        """获取视频文件信息"""
        video_path = self.get_video_path(camera_key, episode_index)
        if not video_path:
            return {}

        return {
            "path": str(video_path),
            "size": video_path.stat().st_size,
            "exists": True
        }
```

### 5.3 FastAPI 服务器 (`server.py`)

```python
"""FastAPI 主服务器"""
from pathlib import Path
from fastapi import FastAPI, HTTPException, Request
from fastapi.responses import FileResponse, StreamingResponse, HTMLResponse
from fastapi.staticfiles import StaticFiles
from fastapi.templating import Jinja2Templates

from .core.dataset_manager import DatasetManager
from .core.video_streamer import VideoStreamer

app = FastAPI(title="LeRobot Dataset Visualizer")

# 配置
DEFAULT_ROOT = Path.home() / ".cache/huggingface/lerobot"
dataset_manager = DatasetManager(DEFAULT_ROOT)

# 静态文件和模板
app.mount("/static", StaticFiles(directory="vis/static"), name="static")
templates = Jinja2Templates(directory="vis/templates")


@app.get("/", response_class=HTMLResponse)
async def index(request: Request):
    """主页面"""
    return templates.TemplateResponse("index.html", {"request": request})


@app.get("/api/datasets")
async def list_datasets():
    """获取数据集列表"""
    return {"datasets": dataset_manager.discover_datasets()}


@app.get("/api/datasets/{user}/{repo}/info")
async def get_dataset_info(user: str, repo: str):
    """获取数据集详情"""
    repo_id = f"{user}/{repo}"
    try:
        return dataset_manager.get_dataset_info(repo_id)
    except FileNotFoundError:
        raise HTTPException(404, f"Dataset not found: {repo_id}")


@app.get("/api/datasets/{user}/{repo}/episodes")
async def list_episodes(user: str, repo: str):
    """获取 Episode 列表"""
    repo_id = f"{user}/{repo}"
    episodes = dataset_manager.get_episodes(repo_id)
    return {"episodes": episodes, "total": len(episodes)}


@app.get("/api/datasets/{user}/{repo}/videos/{camera_key}/{episode_index}")
async def stream_video(user: str, repo: str, camera_key: str, episode_index: int, request: Request):
    """视频流接口，支持 Range 请求"""
    repo_id = f"{user}/{repo}"
    dataset_path = DEFAULT_ROOT / repo_id
    streamer = VideoStreamer(dataset_path)

    video_path = streamer.get_video_path(camera_key, episode_index)
    if not video_path:
        raise HTTPException(404, "Video not found")

    file_size = video_path.stat().st_size

    # 处理 Range 请求
    range_header = request.headers.get("range")
    if range_header:
        start, end = parse_range_header(range_header, file_size)
        return StreamingResponse(
            iter_file(video_path, start, end),
            status_code=206,
            headers={
                "Content-Range": f"bytes {start}-{end}/{file_size}",
                "Accept-Ranges": "bytes",
                "Content-Length": str(end - start + 1),
                "Content-Type": "video/mp4"
```

```javascript
/**
 * LeRobot Dataset Visualizer - 主应用逻辑
 */
class DatasetVisualizer {
    constructor() {
        this.currentDataset = null;
        this.currentEpisode = null;
        this.videoPlayers = {};
        this.chart = null;
        this.frameData = null;
    }

    async init() {
        await this.loadDatasets();
        this.setupEventListeners();
    }

    async loadDatasets() {
        const response = await fetch('/api/datasets');
        const data = await response.json();
        this.renderDatasetList(data.datasets);
    }

    async selectDataset(repoId) {
        this.currentDataset = repoId;
        const [info, episodes] = await Promise.all([
            fetch(`/api/datasets/${repoId}/info`).then(r => r.json()),
            fetch(`/api/datasets/${repoId}/episodes`).then(r => r.json())
        ]);
        this.renderDatasetInfo(info);
        this.renderEpisodeList(episodes.episodes);
    }

    async selectEpisode(episodeIndex) {
        this.currentEpisode = episodeIndex;
        const repoId = this.currentDataset;

        // 获取 episode 详情和帧数据
        const [detail, frames] = await Promise.all([
            fetch(`/api/datasets/${repoId}/episodes/${episodeIndex}`).then(r => r.json()),
            fetch(`/api/datasets/${repoId}/episodes/${episodeIndex}/frames`).then(r => r.json())
        ]);

        this.frameData = frames;
        this.setupVideoPlayers(detail);
        this.setupChart(frames);
    }
}

// 初始化应用
document.addEventListener('DOMContentLoaded', () => {
    const app = new DatasetVisualizer();
    app.init();
});
```

### 6.4 多相机同步播放器 (`static/js/player.js`)

```javascript
/**
 * 多相机同步视频播放器
 */
class MultiCameraPlayer {
    constructor(container, cameras) {
        this.container = container;
        this.cameras = cameras;
        this.videos = {};
        this.isPlaying = false;
        this.currentTime = 0;
    }

    setup(repoId, episodeIndex) {
        this.container.innerHTML = '';
        this.cameras.forEach(camera => {
            const video = document.createElement('video');
            video.src = `/api/datasets/${repoId}/videos/${camera}/${episodeIndex}`;
            video.controls = false;
            video.muted = true;
            this.videos[camera] = video;
            this.container.appendChild(video);
        });
        // 主视频控制其他视频
        const mainVideo = Object.values(this.videos)[0];
        mainVideo.addEventListener('timeupdate', () => this.syncVideos());
    }

    syncVideos() {
        const mainVideo = Object.values(this.videos)[0];
        Object.values(this.videos).slice(1).forEach(v => {
            if (Math.abs(v.currentTime - mainVideo.currentTime) > 0.1) {
                v.currentTime = mainVideo.currentTime;
            }
        });
    }

    play() { Object.values(this.videos).forEach(v => v.play()); }
    pause() { Object.values(this.videos).forEach(v => v.pause()); }
    seek(time) { Object.values(this.videos).forEach(v => v.currentTime = time); }
}
```

### 8.2 技术要点

1. **视频流传输**
   - 使用 HTTP Range 请求支持视频拖拽
   - 直接传输 MP4 文件，无需转码
   - 浏览器原生 `<video>` 标签播放

2. **数据访问**
   - 复用 `lerobot.datasets.utils` 中的工具函数
   - 使用 LRU 缓存减少重复加载
   - 按需加载帧数据，避免内存溢出

3. **前端同步**
   - 多视频同步播放使用 `timeupdate` 事件
   - 图表与视频同步使用 `requestAnimationFrame`
   - 使用 Web Workers 处理大量数据
